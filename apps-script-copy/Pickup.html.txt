    <base target="_top">
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <?!= include('Styles'); ?>
  </head>
  <body>
    <div class="app-shell">
      <header class="topbar">
        <div class="topbar-inner">
          <div class="brand">
            <h1 class="brand-title"><?= appTitle ?> 픽업</h1>
            <p class="brand-subtitle">주문번호별 입금 확인과 준비 상태가 표시됩니다. 5초마다 자동 갱신됩니다.</p>
          </div>
          <nav class="page-actions">
            <a class="button secondary" href="<?= webAppUrl ?>">주문 화면</a>
          </nav>
        </div>
      </header>

      <main class="container">
        <section class="panel">
          <div class="panel-header">
            <h2 class="panel-title">픽업 상태</h2>
            <span id="updatedAt" class="brand-subtitle"></span>
          </div>
          <div class="panel-body">
            <div id="teamTabs" class="segmented" aria-label="팀 선택"></div>
            <div id="pickupBoard" class="pickup-board" aria-live="polite">
              <div class="empty-state">표시할 주문이 없습니다.</div>
            </div>
          </div>
        </section>
      </main>
    </div>

    <script>
      const INITIAL_PICKUP = <?!= initialPickupJson ?>;
      const REQUEST_PARAMS = <?!= requestParamsJson ?>;
      const PERF_LOG_ENABLED = true;
      const CLIENT_BUILD = 'pickup-hybrid-sync-v2';
      const CUSTOMER_FULL_SYNC_INTERVAL_MS = 30000;
      console.log('[BUILD]', { page: 'pickup', build: CLIENT_BUILD });
      const URL_PARAMS = new URLSearchParams(window.location.search);
      const PICKUP_ORDER_ID = URL_PARAMS.get('orderId') || REQUEST_PARAMS.orderId || '';
      const PICKUP_ORDER_TOKEN = URL_PARAMS.get('token') || URL_PARAMS.get('orderToken') || REQUEST_PARAMS.token || REQUEST_PARAMS.orderToken || '';
      const board = document.getElementById('pickupBoard');
      const updatedAt = document.getElementById('updatedAt');
      const teamTabs = document.getElementById('teamTabs');
      const CUSTOMER_ORDER_REFS = loadCustomerOrderRefs();
      const state = {
        teams: [],
        orders: [],
        selectedTeamId: '',
        isRefreshing: false,
        isCustomerFullSyncing: false,
        lastCustomerFullSyncAt: 0,
        customerOrders: CUSTOMER_ORDER_REFS,
        isCustomerMode: CUSTOMER_ORDER_REFS.length > 0
      };

      function perfNow() {
        return window.performance && performance.now ? performance.now() : Date.now();
      }

      function perfLog(name, startedAt, extra = {}) {
        if (!PERF_LOG_ENABLED) return;
        const event = {
          kind: 'perf',
          page: 'pickup',
          build: CLIENT_BUILD,
          name,
          ms: Math.round((perfNow() - startedAt) * 10) / 10,
          ...extra
        };
        console.log('[PERF]', event);
      }

      function callServer(method, payload) {
        const startedAt = perfNow();
        return new Promise((resolve, reject) => {
          google.script.run
            .withSuccessHandler((result) => {
              perfLog(`server.${method}`, startedAt, { ok: true });
              resolve(result);
            })
            .withFailureHandler((error) => {
              perfLog(`server.${method}`, startedAt, { ok: false, error: error.message || String(error) });
              reject(error);
            })[method](payload);
        });
      }

      function escapeHtml(value) {
        return String(value ?? '')
          .replace(/&/g, '&amp;')
          .replace(/</g, '&lt;')
          .replace(/>/g, '&gt;')
          .replace(/"/g, '&quot;')
          .replace(/'/g, '&#039;');
      }

      function statusText(status) {
        if (status === 'PAYMENT_PENDING') return '입금 대기';
        if (status === 'READY') return '준비 완료';
        if (status === 'PAID') return '준비 중';
        if (status === 'PAYMENT_CHECKING') return '입금 확인 중';
        if (status === 'COMPLETE') return '수령 완료';
        if (status === 'PAYMENT_ISSUE') return '입금 확인 필요';
        return '상태 확인 중';
      }

      function renderTeamTabs() {
        if (state.isCustomerMode) {
          teamTabs.innerHTML = '';
          teamTabs.classList.add('is-hidden');
          return;
        }
        teamTabs.classList.remove('is-hidden');
        if (!state.teams.length) {
          teamTabs.innerHTML = '';
          return;
        }
        const allActive = !state.selectedTeamId;
        teamTabs.innerHTML = `
          <button class="${allActive ? 'is-active' : ''}" type="button" data-team-id="">전체</button>
          ${state.teams.map((team) => `
            <button class="${team.teamId === state.selectedTeamId ? 'is-active' : ''}" type="button" data-team-id="${escapeHtml(team.teamId)}">
              ${escapeHtml(team.teamName)}
            </button>
          `).join('')}
        `;
      }

      function renderOrders() {
        const orders = state.selectedTeamId
          ? state.orders.filter((order) => order.teamId === state.selectedTeamId)
          : state.orders;

        if (!orders.length) {
          board.innerHTML = '<div class="empty-state">표시할 주문이 없습니다.</div>';
          return;
        }

        board.innerHTML = orders.map((order) => `
          <article class="pickup-ticket ${order.status === 'READY' ? 'is-ready' : 'is-preparing'}">
            <div class="pickup-ticket-main">
              <strong>${escapeHtml(order.orderId)}</strong>
              <span>${escapeHtml(order.teamName)} · ${escapeHtml(statusText(order.status))}</span>
              <small>${escapeHtml(order.itemsText || '')}</small>
            </div>
            ${order.status === 'READY' && order.teamId !== 'multi' ? `
              <button class="button full" type="button" data-action="complete" data-order-id="${escapeHtml(order.orderId)}" data-team-id="${escapeHtml(order.teamId)}">
                수령 완료
              </button>
            ` : ''}
          </article>
        `).join('');
      }

      function applyPickupSnapshot(result) {
        const startedAt = perfNow();
        updatedAt.textContent = result.updatedAt ? `갱신 ${result.updatedAt}` : '';
        state.teams = result.teams || state.teams || [];
        state.orders = result.orders || [];
        renderTeamTabs();
        renderOrders();
        perfLog('applyPickupSnapshot', startedAt, { orders: state.orders.length });
      }

      function orderToSnapshot(order) {
        if (!order || !order.orderId) return null;
        return {
          orders: [{
            orderId: order.orderId,
            teamId: order.teamId || 'multi',
            teamName: order.teamName || '',
            status: order.status,
            statusLabel: order.statusLabel,
            itemsText: order.itemsText || ''
          }],
          teams: [],
          updatedAt: order.updatedAt || ''
        };
      }

      function toOrderRef(order) {
        return {
          orderId: String(order.orderId || ''),
          orderToken: String(order.orderToken || '')
        };
      }

      function customerKeyFromOrder(order) {
        const name = String(order?.pickupName || '').trim().replace(/\s+/g, '').toLowerCase();
        const phone = String(order?.phone || '').replace(/\D/g, '');
        return name && phone ? `${name}:${phone}` : '';
      }

      function loadCustomerOrderRefs() {
        let orders = [];
        try {
          orders = JSON.parse(sessionStorage.getItem('missionBazaarPickupOrders') || '[]');
          if (!Array.isArray(orders)) orders = [];
        } catch (error) {
          orders = [];
        }

        const refs = [];
        if (PICKUP_ORDER_ID && PICKUP_ORDER_TOKEN) {
          const savedOrder = orders.find((order) =>
            String(order.orderId || '') === PICKUP_ORDER_ID &&
            String(order.orderToken || '') === PICKUP_ORDER_TOKEN
          );
          refs.push(savedOrder ? toOrderRef(savedOrder) : {
            orderId: PICKUP_ORDER_ID,
            orderToken: PICKUP_ORDER_TOKEN
          });
        }

        orders.map(toOrderRef).forEach((ref) => refs.push(ref));

        const seen = new Set();
        const uniqueRefs = refs.filter((ref) => {
          const key = `${ref.orderId}:${ref.orderToken}`;
          if (!ref.orderId || !ref.orderToken || seen.has(key)) return false;
          seen.add(key);
          return true;
        }).slice(0, 20);

        return uniqueRefs;
      }

      function loadSavedPickupSnapshot() {
        if (!state.isCustomerMode) return null;
        let savedOrders = [];
        try {
          savedOrders = JSON.parse(sessionStorage.getItem('missionBazaarPickupOrders') || '[]');
          if (!Array.isArray(savedOrders)) savedOrders = [];
        } catch (error) {
          savedOrders = [];
        }

        const savedById = new Map(savedOrders.map((order) => [String(order.orderId || ''), order]));
        const seedOrder = PICKUP_ORDER_ID ? savedById.get(PICKUP_ORDER_ID) : savedOrders[0];
        const customerKey = customerKeyFromOrder(seedOrder);
        const refIds = new Set(state.customerOrders.map((ref) => String(ref.orderId || '')));
        const localCards = savedOrders
          .filter((order) => refIds.has(String(order.orderId || '')))
          .filter((order) => {
            if (!customerKey) return true;
            return customerKeyFromOrder(order) === customerKey;
          })
          .map((order) => (orderToSnapshot(order) || {}).orders || [])
          .flat();
        const prefetched = loadPrefetchedPickupSnapshot(customerKey);
        const cards = mergeOrders((prefetched && prefetched.orders) || [], localCards);

        if (!cards.length) return null;
        return {
          orders: cards,
          teams: (prefetched && prefetched.teams) || [],
          updatedAt: (prefetched && prefetched.updatedAt) || cards[0].updatedAt || ''
        };
      }

      function loadPrefetchedPickupSnapshot(customerKey) {
        if (!customerKey) return null;
        try {
          const saved = JSON.parse(sessionStorage.getItem('missionBazaarPickupSnapshot') || 'null');
          if (!saved || saved.customerKey !== customerKey || !saved.result || !Array.isArray(saved.result.orders)) {
            return null;
          }
          return saved.result;
        } catch (error) {
          return null;
        }
      }

      function mergeOrders(serverOrders, localOrders) {
        const map = new Map();
        const serverOrderIds = new Set(serverOrders.map((order) => String(order.orderId || '')));
        localOrders.forEach((order) => {
          if (serverOrderIds.has(String(order.orderId || ''))) return;
          map.set(`${order.orderId}:${order.teamId}`, order);
        });
        serverOrders.forEach((order) => {
          map.set(`${order.orderId}:${order.teamId}`, order);
        });
        return Array.from(map.values());
      }

      function primaryCustomerRef() {
        return state.customerOrders.find((ref) => ref.orderId && ref.orderToken) || null;
      }

      function applyPrimaryOrderSnapshot(result) {
        const mergedOrders = mergeOrders((result && result.orders) || [], state.orders || []);
        applyPickupSnapshot({
          orders: mergedOrders,
          teams: (result && result.teams && result.teams.length) ? result.teams : state.teams,
          updatedAt: (result && result.updatedAt) || ''
        });
      }

      async function refreshCustomerOrdersFull(force = false) {
        if (!state.isCustomerMode || state.isCustomerFullSyncing) return;
        const now = perfNow();
        if (!force && now - state.lastCustomerFullSyncAt < CUSTOMER_FULL_SYNC_INTERVAL_MS) return;

        state.isCustomerFullSyncing = true;
        state.lastCustomerFullSyncAt = now;
        const startedAt = perfNow();
        try {
          const result = await callServer('getPickupOrdersForCustomer', {
            orders: state.customerOrders
          });
          applyPickupSnapshot({
            ...result,
            orders: mergeOrders((result && result.orders) || [], state.orders || [])
          });
        } finally {
          perfLog('refreshCustomerOrdersFull.total', startedAt, {
            orders: state.orders.length
          });
          state.isCustomerFullSyncing = false;
        }
      }

      function queueCustomerFullSync(force = false) {
        if (!state.isCustomerMode) return;
        setTimeout(() => refreshCustomerOrdersFull(force).catch(() => {}), 0);
      }

      async function refreshPickup(includeTeams = false) {
        if (state.isRefreshing) return;
        const startedAt = perfNow();
        state.isRefreshing = true;
        try {
          if (state.isCustomerMode) {
            const ref = primaryCustomerRef();
            if (!ref) return;
            applyPrimaryOrderSnapshot(await callServer('refreshPickupOrder', {
              orderId: ref.orderId,
              orderToken: ref.orderToken
            }));
            queueCustomerFullSync(false);
          } else {
            applyPickupSnapshot(await callServer(includeTeams ? 'getPickupOrders' : 'refreshPickupOrders'));
          }
        } finally {
          perfLog('refreshPickup.total', startedAt, {
            customerMode: state.isCustomerMode,
            includeTeams,
            orders: state.orders.length
          });
          state.isRefreshing = false;
        }
      }

      async function completePickup(orderId, teamId) {
        if (!confirm(`${orderId} 주문을 수령 완료 처리할까요?`)) return;
        const result = await callServer('completePickup', { orderId, teamId });
        if (state.isCustomerMode) {
          await refreshPickup();
        } else {
          applyPickupSnapshot(result);
        }
      }

      teamTabs.addEventListener('click', (event) => {
        const button = event.target.closest('button[data-team-id]');
        if (!button) return;
        state.selectedTeamId = button.dataset.teamId;
        renderTeamTabs();
        renderOrders();
      });

      board.addEventListener('click', (event) => {
        const button = event.target.closest('button[data-action="complete"]');
        if (!button) return;
        completePickup(button.dataset.orderId, button.dataset.teamId).catch((error) => alert(error.message || error));
      });

      const savedPickupSnapshot = loadSavedPickupSnapshot();
      if (state.isCustomerMode && savedPickupSnapshot) {
        perfLog('initialSavedSnapshot', perfNow(), { orders: savedPickupSnapshot.orders.length });
        applyPickupSnapshot(savedPickupSnapshot);
        queueCustomerFullSync(true);
        setTimeout(() => refreshPickup().catch(() => {}), 5000);
      } else if (state.isCustomerMode) {
        refreshPickup().then(() => queueCustomerFullSync(true)).catch((error) => {
          board.innerHTML = `<div class="empty-state">${escapeHtml(error.message || error)}</div>`;
        });
      } else if (INITIAL_PICKUP && !INITIAL_PICKUP.error) {
        applyPickupSnapshot(INITIAL_PICKUP);
      } else if (INITIAL_PICKUP && INITIAL_PICKUP.error) {
        board.innerHTML = `<div class="empty-state">${escapeHtml(INITIAL_PICKUP.error)}</div>`;
        refreshPickup(true).catch(() => {});
      } else {
        refreshPickup(true).catch((error) => {
          board.innerHTML = `<div class="empty-state">${escapeHtml(error.message || error)}</div>`;
        });
      }
      setInterval(() => refreshPickup().catch(() => {}), 5000);
    </script>
  </body>
